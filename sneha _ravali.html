flank = 10000  # 10 kb each side
hits = pd.read_csv(summary_csv, dtype=str)

def revcomp_if_needed(seq, strand):
    if strand.strip() == "-":
        return seq.reverse_complement()
    return seq

for _, r in hits.iterrows():
    isolate = r["Isolate"]
    gene = r["Gene"]
    contig = r["Contig"]
    start = int(r["Start"])
    end = int(r["End"])
    strand = r["Strand"]

    assembly_path = os.path.join(assemblies_dir, f"{isolate}_assembly.fasta")

    contig_record = None
    for record in SeqIO.parse(assembly_path, "fasta"):
        if record.id == contig:
            contig_record = record
            break

    if contig_record is None:
        print(f"Missing contig {contig} in {assembly_path}")
        continue

    s = min(start, end)
    e = max(start, end)

    left = max(0, (s - 1) - flank)  # start is 1-based
    right = min(len(contig_record.seq), e + flank)

    context_seq = contig_record.seq[left:right]
    context_seq = revcomp_if_needed(context_seq, strand)

    header = f"{isolate}|{contig}|{gene}|{s}-{e}|flank{flank}|strand{strand}"
    out_path = os.path.join(out_dir, f"{isolate}_{contig}_{gene}_context_{flank//1000}kb.fasta")

    SeqIO.write(SeqRecord(context_seq, id=header, description=""), out_path, "fasta")
    print("Wrote:", out_path)

    return gb_seq_map.get(str(fasta_rec.seq), None), fasta_rec.id

    # weighted jaccard on kmer counts
    keys = set(c1) | set(c2)

    union = sum(max(c1.get(k,0), c2.get(k,0)) for k in keys)

# --- Similarity functions ---
def kmers(seq, k=21):
    s = str(seq).upper()
    return Counter(s[i:i+k] for i in range(len(s)-k+1))

def wjaccard(a, b):
    keys = set(a) | set(b)
    inter = sum(min(a.get(k,0), b.get(k,0)) for k in keys)
    union = sum(max(a.get(k,0), b.get(k,0)) for k in keys)
    return inter/union if union else 0.0

# --- Load FASTAs robustly ---
def load_fasta_folder(folder, source_label):
    loaded = []
    skipped = []
    for f in sorted(os.listdir(folder)):
        if not f.endswith(".fasta"):
            continue
        path = os.path.join(folder, f)
        try:
            recs = list(SeqIO.parse(path, "fasta-pearson"))  # tolerant parser
            if not recs:
                skipped.append(f)
                continue
            loaded.append((f, recs[0].seq))
        except Exception:
            skipped.append(f)
    print(f"{source_label} loaded:", len(loaded), "| skipped:", len(skipped))
    if skipped:
        print("First skipped files:", skipped[:10])
    return loaded

globals_ = load_fasta_folder(global_dir, "GLOBAL")
